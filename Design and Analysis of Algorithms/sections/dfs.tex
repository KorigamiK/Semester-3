\section{Depth First Search}

\subsection{Description}
The depth-first search (DFS) algorithm starts with the initial node of graph G and goes deeper until we find the goal
node or the node with no children.

\subsection{Algorithm}

% 1.	DFS(G, v)
% 2.	mark v as visited
% 3.	for each unvisited neighbor w of v
% 4.	DFS(G, w)

\begin{algorithm}[H]
    \caption{Depth First Search}
    \begin{algorithmic}[1]
        \Procedure{DFS}{$G, v$}
        \State mark $v$ as visited
        \For{each unvisited neighbor $w$ of $v$}
        \State DFS($G, w$)
        \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
